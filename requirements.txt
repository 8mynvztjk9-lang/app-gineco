# Librerías que necesita el proyecto.
# Se instalan con:  pip install -r requirements.txt

fastapi              # el framework web (crea la API y las páginas)
uvicorn[standard]    # el servidor que ejecuta la aplicación
sqlalchemy           # habla con la base de datos SQLite de forma sencilla
jinja2               # genera las páginas HTML
python-multipart     # permite recibir datos de formularios (login, etc.)
itsdangerous         # firma las cookies de sesión (mantiene el login)
